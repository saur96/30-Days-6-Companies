# 396. Rotate Function

[![Problem Link](../../assets/lc.svg)](https://leetcode.com/problems/rotate-function/)

### Solution
```
class Solution {
public:
    int maxRotateFunction(vector<int>& nums) {
        long long sum=accumulate(nums.begin(),nums.end(),0);
        int n=nums.size();
        long long ans=0;
        for(long long i=0;i<n;i++)
        {
            ans+=i*nums[i];
        }
        long long temp=ans;
        for(int i=0;i<n;i++)
        {
            long long f=nums[n-i-1];
            temp=(temp+sum)-(f*(n));
            ans=max(ans,temp);
        }
        return ans;
    }
};


```
